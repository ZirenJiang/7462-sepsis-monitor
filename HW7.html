get_observation_data <- function(data, n, column_suffix) {
  data %>%
    group_by(PatientID) %>%
    slice_max(order_by = obsTime, n = n) %>%
    select(-c(ICULOS, SepsisLabel, obsTime)) %>%
    rename_at(vars(HR, Temp, Resp),
              list(~paste0(., "_", column_suffix)))
}
